  if (!profile?.joinedDate) return 1;
  const joined = new Date(profile.joinedDate);
  const now = new Date();
  const diffMs = now.getTime() - joined.getTime();
  return Math.max(1, Math.ceil(diffMs / (1000 * 60 * 60 * 24)));
}, [profile?.joinedDate]);
